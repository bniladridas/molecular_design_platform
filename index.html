Code Example

# Import molecular design platform modules
from molecular_design.core import Molecule
from molecular_design.prediction import PropertyCalculator
from molecular_design.optimization import Optimizer

# Load a molecule from SMILES notation
molecule = Molecule.from_smiles("CC(=O)OC1=CC=CC=C1C(=O)O")

# Calculate molecular properties
properties = PropertyCalculator.calculate(molecule, 
                                         properties=["logP", "TPSA", "HBA", "HBD"])
print(f"LogP: {properties['logP']}")
print(f"TPSA: {properties['TPSA']} Å²")

# Define optimization objective
def objective_function(mol):
    props = PropertyCalculator.calculate(mol, properties=["logP", "TPSA"])
    # Optimize for oral bioavailability
    score = 0
    if 1 < props["logP"] < 5:
        score += 1
    if props["TPSA"] < 140:
        score += 1
    return score

# Run optimization
optimizer = Optimizer(objective_function)
optimized_molecules = optimizer.optimize(molecule, 
                                        num_iterations=100,
                                        population_size=50)

# Get the best molecule
best_molecule = optimized_molecules[0]
print(f"Best molecule SMILES: {best_molecule.to_smiles()}")
